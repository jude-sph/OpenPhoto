# Phase 4.3 — Faces & People (full management) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Detect faces in every photo (Apple Vision), embed each cropped face (`VNGenerateImageFeaturePrint`), cluster look-alike faces, and provide **full People management** — name a cluster → Person, merge two people, split a cluster, reassign an individual face, with per-face confidence shown — writing **named/confirmed** assignments to **MWG face regions in XMP sidecars** as fixed anchors. A **person** filter folds into the 4.2 unified Search.

**Architecture:** A third `DerivationStage` (`FaceDerivationStage` over `FaceStage`, Vision) writes one `faces` row per detected face into new rebuildable catalog tables `faces`/`people` (migration **v8**; rects + Float16 feature-print embeddings + confidence; `personID`/`name` mirror human decisions). A **pure** `FaceClusterer` (cosine, greedy agglomerative) groups the unassigned set on demand. A **People** view names/merges/splits/reassigns, pairing every confirmed catalog change with a sidecar **`mwg-rs:Regions`** write. `SearchFilters.person` + a `faces` join in `Catalog.structuredFilter` add the person lane to Search.

**Tech Stack:** Swift 6 / SwiftUI / SwiftPM **Command Line Tools only** (`swift build` / `swift test`, no Xcode), Apple **Vision** (`VNDetectFaceRectanglesRequest` + `VNGenerateImageFeaturePrint`), GRDB (SQLite + `json_each`), `XMLDocument` (XMP), Swift Testing.

`Sources/OpenPhotoCore/Catalog/Catalog+Faces.swift` (reuse the Float16 pack/unpack idiom from `Catalog+Embeddings.swift` — keep a local copy or hoist to a shared internal helper; do not duplicate the *public* surface):
```swift
import Foundation
import CoreGraphics
import GRDB

public struct FaceRow: Sendable, Equatable {
    public var id: Int64?
    public var hash: String
    public var rect: CGRect            // Vision normalized boundingBox (bottom-left origin)
    public var embedding: [Float]
    public var confidence: Float
    public var source: String          // "auto" | "confirmed"
    public var personID: Int64?
    public init(id: Int64?, hash: String, rect: CGRect, embedding: [Float],
                confidence: Float, source: String, personID: Int64?) {
        self.id = id; self.hash = hash; self.rect = rect; self.embedding = embedding
        self.confidence = confidence; self.source = source; self.personID = personID
    }
}

public struct PersonRow: Sendable, Equatable {
    public let id: Int64
    public let name: String
    public let faceCount: Int
    public let representativeFaceID: Int64?   // highest-confidence confirmed face
}

extension Catalog {
    private static func packF16(_ v: [Float]) -> Data { /* identical to Catalog+Embeddings */ }
    private static func unpackF16(_ d: Data, dim: Int) -> [Float] { /* identical */ }

    /// Low-level insert; returns the new ids in order.
    public func insertFaces(_ rows: [FaceRow]) throws -> [Int64] {
        try dbQueue.write { db in
            var ids: [Int64] = []
            for r in rows {
                try db.execute(sql: """
                    INSERT INTO faces (hash, rectX, rectY, rectW, rectH, embedding, dim,
                                       personID, confidence, source)
                    VALUES (?,?,?,?,?,?,?,?,?,?)
                    """, arguments: [r.hash, r.rect.minX, r.rect.minY, r.rect.width, r.rect.height,
                                     Self.packF16(r.embedding), r.embedding.count,
                                     r.personID, r.confidence, r.source])
                ids.append(db.lastInsertedRowID)
            }
            return ids
        }
    }

    /// Idempotent re-detection: replace the asset's AUTO faces, KEEP confirmed (and their regions).
    public func replaceFaces(forHash hash: String, with rows: [FaceRow]) throws {
        try dbQueue.write { db in
            try db.execute(sql: "DELETE FROM faces WHERE hash = ? AND source = 'auto'", arguments: [hash])
            for r in rows {
                try db.execute(sql: """
                    INSERT INTO faces (hash, rectX, rectY, rectW, rectH, embedding, dim,
                                       personID, confidence, source)
                    VALUES (?,?,?,?,?,?,?,NULL,?,?)
                    """, arguments: [hash, r.rect.minX, r.rect.minY, r.rect.width, r.rect.height,
                                     Self.packF16(r.embedding), r.embedding.count,
                                     r.confidence, r.source])
            }
        }
    }

    public func faces(forHash hash: String) throws -> [FaceRow] { /* SELECT … WHERE hash = ? */ }
    public func faces(forPerson personID: Int64) throws -> [FaceRow] { /* WHERE personID = ? */ }

    /// (id, vector) for every detected-but-unassigned auto face — the clusterer's input.
    public func unassignedFacesWithEmbeddings() throws -> [(id: Int64, vector: [Float])] {
        try dbQueue.read { db in
            try Row.fetchAll(db, sql: """
                SELECT id, dim, embedding FROM faces WHERE personID IS NULL AND source = 'auto'
                """).map { (Int64($0["id"]), Self.unpackF16($0["embedding"], dim: $0["dim"])) }
        }
    }

    public func createPerson(name: String) throws -> Int64 {
        try dbQueue.write { db in
            try db.execute(sql: "INSERT INTO people (name, createdAtMs) VALUES (?, ?)",
                           arguments: [name, Int64(Date().timeIntervalSince1970 * 1000)])
            return db.lastInsertedRowID
        }
    }

    public func people() throws -> [PersonRow] {
        // LEFT JOIN faces; COUNT(*) per person; representative = the highest-confidence face id.
        try dbQueue.read { db in
            try Row.fetchAll(db, sql: """
                SELECT p.id, p.name, COUNT(f.id) AS cnt,
                       (SELECT f2.id FROM faces f2 WHERE f2.personID = p.id
                        ORDER BY f2.confidence DESC LIMIT 1) AS rep
                FROM people p LEFT JOIN faces f ON f.personID = p.id
                GROUP BY p.id ORDER BY cnt DESC, p.name
                """).map { PersonRow(id: $0["id"], name: $0["name"], faceCount: $0["cnt"],
                                     representativeFaceID: $0["rep"]) }
        }
    }

    /// Assign faces to a person + flip them to "confirmed" (the App pairs this with a sidecar write).
    public func assignFaces(_ ids: [Int64], to personID: Int64) throws {
        guard !ids.isEmpty else { return }
        try dbQueue.write { db in
            let marks = databaseQuestionMarks(count: ids.count)
            try db.execute(sql: """
                UPDATE faces SET personID = ?, source = 'confirmed' WHERE id IN (\(marks))
                """, arguments: StatementArguments([personID] + ids))
        }
    }

    /// Reassign one face. nil → unassign (personID NULL, source back to 'auto').
    public func reassignFace(_ id: Int64, to personID: Int64?) throws {
        try dbQueue.write { db in
            if let personID {
                try db.execute(sql: "UPDATE faces SET personID = ?, source = 'confirmed' WHERE id = ?",
                               arguments: [personID, id])
            } else {
                try db.execute(sql: "UPDATE faces SET personID = NULL, source = 'auto' WHERE id = ?",
                               arguments: [id])
            }
        }
    }

    /// Move src's faces to dst, delete src person.
    public func mergePerson(_ src: Int64, into dst: Int64) throws {
        try dbQueue.write { db in
            try db.execute(sql: "UPDATE faces SET personID = ? WHERE personID = ?", arguments: [dst, src])
            try db.execute(sql: "DELETE FROM people WHERE id = ?", arguments: [src])
        }
    }

    /// Delete a person; its faces revert to unassigned/auto (re-enter the clusterable pool).
    public func deletePerson(_ id: Int64) throws {
        try dbQueue.write { db in
            try db.execute(sql: "UPDATE faces SET personID = NULL, source = 'auto' WHERE personID = ?",
                           arguments: [id])
            try db.execute(sql: "DELETE FROM people WHERE id = ?", arguments: [id])
        }
    }
}
```

```swift
import Foundation
import Vision
import CoreGraphics
import ImageIO

public struct DetectedFace: Sendable {
    public let rect: CGRect      // Vision normalized boundingBox (bottom-left origin, 0…1)
    public let embedding: [Float]
    public let confidence: Float
}

/// On-device human-face detection + per-face embedding (Apple Vision). Headless + synchronous;
/// callers run it off the main actor. Human faces only (no pets).
public enum FaceStage {
    public static let id = "faces"

    /// Detect faces; per face crop the bounding box and run VNGenerateImageFeaturePrint on the crop.
    /// Returns one DetectedFace per face ([] for a photo with no faces); nil only if the SOURCE image
    /// can't be decoded (so the runner records a failure, retry-capped).
    public static func detect(in url: URL) -> [DetectedFace]? {
        guard let src = CGImageSourceCreateWithURL(url as CFURL, nil),
              let cg = CGImageSourceCreateImageAtIndex(src, 0, nil) else { return nil }
        let req = VNDetectFaceRectanglesRequest()
        let handler = VNImageRequestHandler(cgImage: cg, options: [:])
        do { try handler.perform([req]) } catch { return nil }
        let observations = req.results ?? []
        let w = CGFloat(cg.width), h = CGFloat(cg.height)
        var out: [DetectedFace] = []
        for obs in observations {
            // Vision boundingBox is normalized, bottom-left origin. Crop in PIXEL space (top-left).
            let bb = obs.boundingBox
            let px = CGRect(x: bb.minX * w, y: (1 - bb.maxY) * h, width: bb.width * w, height: bb.height * h)
                .insetBy(dx: -bb.width * w * 0.1, dy: -bb.height * h * 0.1)   // small pad
                .intersection(CGRect(x: 0, y: 0, width: w, height: h))
            guard let crop = cg.cropping(to: px) else { continue }
            guard let vec = featurePrint(of: crop) else { continue }
            out.append(DetectedFace(rect: bb, embedding: vec, confidence: obs.confidence))
        }
        return out
    }

    private static func featurePrint(of cg: CGImage) -> [Float]? {

        do { try handler.perform([req]) } catch { return nil }
        guard let obs = req.results?.first as? VNFeaturePrintObservation else { return nil }
        // Feature-print is Float32; read elementCount floats from .data.
        let count = obs.elementCount
        var vec = [Float](repeating: 0, count: count)
        obs.data.withUnsafeBytes { (raw: UnsafeRawBufferPointer) in
            let f = raw.bindMemory(to: Float.self)
            for i in 0..<min(count, f.count) { vec[i] = f[i] }
        }
        return vec
    }
}
```

```swift
import Foundation

/// Pure cosine-distance agglomerative clustering of face feature-print vectors (the unit-tested
/// heart — no Vision/Core ML dependency). VNGenerateImageFeaturePrint vectors are NOT normalized
/// and vary in magnitude, so we use COSINE distance (direction only) rather than raw L2.
public enum FaceClusterer {
    /// Greedy single-link agglomerative grouping. `threshold` = max cosine distance (1 − cos sim)
    /// for two faces to share a cluster. Clusters ordered largest-first; each cluster ordered by
    /// input order. Singletons are 1-element clusters. Deterministic.
    public static func cluster(_ items: [(id: Int64, vector: [Float])],
                               threshold: Double) -> [[Int64]] {
        guard !items.isEmpty else { return [] }
        // Determine the dominant dim; vectors of a different dim are kept but only ever compared to
        // same-dim members (cross-dim "distance" is treated as infinite → they never join).
        let normalized: [(id: Int64, v: [Float], dim: Int)] = items.map { item in
            let n = l2norm(item.vector)
            let unit = n > 0 ? item.vector.map { $0 / n } : item.vector
            return (item.id, unit, item.vector.count)
        }
        var clusters: [[Int]] = []   // indices into `normalized`
        for i in normalized.indices {
            var best = -1
            for (ci, members) in clusters.enumerated() {
                // single-link: nearest member within threshold
                let near = members.contains { j in cosineDistance(normalized[i], normalized[j]) <= threshold }
                if near { best = ci; break }
            }
            if best >= 0 { clusters[best].append(i) } else { clusters.append([i]) }
        }
        return clusters
            .map { idxs in idxs.map { normalized[$0].id } }
            .sorted { $0.count > $1.count }
    }

    private static func l2norm(_ v: [Float]) -> Float {
        var s: Float = 0; for x in v { s += x * x }; return s.squareRoot()
    }
    private static func cosineDistance(_ a: (id: Int64, v: [Float], dim: Int),
                                       _ b: (id: Int64, v: [Float], dim: Int)) -> Double {
        guard a.dim == b.dim else { return .infinity }   // never join cross-dim
        var dot: Float = 0
        for k in 0..<a.dim { dot += a.v[k] * b.v[k] }     // unit vectors → dot == cosine
        return 1.0 - Double(dot)
    }
}
```
> Cosine on L2-normalized vectors = `1 − dot`, matching 4.2's machinery. Greedy single-link is O(n·k); the input is the *unassigned* set, so it's bounded. Distance is symmetric and deterministic ⇒ the tests pin behavior exactly.

```swift
import Foundation
import CoreGraphics

/// A human-confirmed face region as it appears in a sidecar. Stored in the catalog's Vision frame
/// (normalized boundingBox, bottom-left origin, lower-left corner) and written to XMP as an MWG
/// `Area` (center + size, TOP-left origin). The two reps are converted symmetrically here.
public struct FaceRegion: Sendable, Equatable {
    public let name: String
    public let visionRect: CGRect          // Vision normalized boundingBox (bottom-left origin)
    public init(name: String, visionRect: CGRect) { self.name = name; self.visionRect = visionRect }

    public struct MWGArea: Sendable, Equatable {   // normalized, top-left origin, center-based
        public let cx: Double, cy: Double, w: Double, h: Double
    }

    /// Vision lower-left corner (bottom-left origin) → MWG center (top-left origin).
    public static func mwgArea(fromVision r: CGRect) -> MWGArea {
        let cx = Double(r.midX)
        let cy = 1.0 - Double(r.midY)           // flip y to top-left origin
        return MWGArea(cx: cx, cy: cy, w: Double(r.width), h: Double(r.height))
    }
    /// MWG center (top-left origin) → Vision lower-left corner (bottom-left origin).
    public static func visionRect(fromMWG a: MWGArea) -> CGRect {
        let midYBottom = 1.0 - a.cy
        return CGRect(x: a.cx - a.w / 2, y: midYBottom - a.h / 2, width: a.w, height: a.h)
    }
}
```
