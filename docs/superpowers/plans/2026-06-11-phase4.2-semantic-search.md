# Phase 4.2 — Semantic embeddings + unified Search Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an on-device MobileCLIP image-embedding derivation stage and a unified Search surface (semantic + OCR + tag/caption lanes, with click-to-narrow structured filters), built on the 4.1 derivation-pipeline spine.

**Architecture:** A second `DerivationStage` (`EmbedStage`, Core ML) writes an L2-normalized vector per photo into a new rebuildable catalog table `embeddings` (v6). A `SemanticIndex` (Accelerate brute-force cosine) + a pure `SearchRanker` + SQL filter/text lanes on `Catalog` power `AppState.runSearch()`, rendered in the existing `MediaTile` grid under a new `SidebarItem.search`.

**Tech Stack:** Swift 6 / SwiftUI / SwiftPM **Command Line Tools only** (`swift build` / `swift test`, no Xcode), Core ML + Accelerate, GRDB (SQLite + FTS5 + `json_each`), Swift Testing.

```swift
import Foundation
import GRDB

extension Catalog {
    /// Pack/unpack L2-normalized vectors as little-endian Float16 (half the footprint; cosine==dot).
    private static func packFloat16(_ v: [Float]) -> Data {
        var data = Data(capacity: v.count * 2)
        for f in v { var h = Float16(f); withUnsafeBytes(of: &h) { data.append(contentsOf: $0) } }
        return data
    }
    private static func unpackFloat16(_ data: Data, dim: Int) -> [Float] {
        var out = [Float](); out.reserveCapacity(dim)
        data.withUnsafeBytes { (raw: UnsafeRawBufferPointer) in
            let halves = raw.bindMemory(to: Float16.self)
            for i in 0..<min(dim, halves.count) { out.append(Float(halves[i])) }
        }
        return out
    }

    /// Replace the embedding for an asset (idempotent upsert).
    public func upsertEmbedding(hash: String, model: String, dim: Int, vector: [Float]) throws {
        let blob = Self.packFloat16(vector)
        try dbQueue.write { db in
            try db.execute(sql: """
                INSERT INTO embeddings (hash, model, dim, vector) VALUES (?, ?, ?, ?)
                ON CONFLICT(hash) DO UPDATE SET model=excluded.model, dim=excluded.dim,
                    vector=excluded.vector
                """, arguments: [hash, model, dim, blob])
        }
    }

    public func embedding(forHash hash: String) -> (model: String, dim: Int, vector: [Float])? {
        try? dbQueue.read { db -> (String, Int, [Float])? in
            guard let row = try Row.fetchOne(db,
                sql: "SELECT model, dim, vector FROM embeddings WHERE hash = ?", arguments: [hash])
            else { return nil }
            let dim: Int = row["dim"]
            return (row["model"], dim, Self.unpackFloat16(row["vector"], dim: dim))
        }.flatMap { $0 }
    }

    public func embeddingCount() throws -> Int {
        try dbQueue.read { db in try Int.fetchOne(db, sql: "SELECT COUNT(*) FROM embeddings") ?? 0 }
    }

    /// All (hash, vector) pairs for the in-memory index, for the current model only.
    public func allEmbeddings(model: String) throws -> [(hash: String, dim: Int, vector: [Float])] {
        try dbQueue.read { db in
            try Row.fetchAll(db, sql: "SELECT hash, dim, vector FROM embeddings WHERE model = ?",
                             arguments: [model]).map { row in
                let dim: Int = row["dim"]
                return (row["hash"], dim, Self.unpackFloat16(row["vector"], dim: dim))
            }
        }
    }

    /// On a model swap, drop stale-model embeddings + their `embed` jobs so the pipeline re-derives.
    /// No-op in v1's single-model world, but the mechanism must exist.
    public func reconcileEmbeddingModel(current: String) throws {
        try dbQueue.write { db in
            let stale = try String.fetchAll(db,
                sql: "SELECT hash FROM embeddings WHERE model <> ?", arguments: [current])
            guard !stale.isEmpty else { return }
            try db.execute(sql: "DELETE FROM embeddings WHERE model <> ?", arguments: [current])
            for h in stale {
                try db.execute(sql: "DELETE FROM derivation_jobs WHERE hash = ? AND stage = 'embed'",
                               arguments: [h])
            }
        }
    }
}
```

```swift
import Foundation
import CoreML

/// On-device CLIP-class embedder. Image encoder runs per-asset in the background pipeline;
/// text encoder runs at query time. Vectors are L2-normalized so cosine == dot product.
public final class EmbedStage: @unchecked Sendable {
    public let modelID = "mobileclip_s2"     // matches the bundled artifact (see the spike doc)
    public let dim: Int = 512                 // variant-dependent; set per the spike doc

    private let lock = NSLock()
    private var imageModel: MLModel?
    private var textModel: MLModel?
    private let tokenizer = CLIPTokenizer()

    public init() {}

    private func model(resource: String, cached: inout MLModel?) -> MLModel? {
        lock.lock(); defer { lock.unlock() }
        if let cached { return cached }
        guard let url = Bundle.module.url(forResource: resource, withExtension: "mlpackage"),
              let compiled = try? MLModel.compileModel(at: url) else { return nil }
        let cfg = MLModelConfiguration(); cfg.computeUnits = .all
        let m = try? MLModel(contentsOf: compiled, configuration: cfg)
        cached = m; return m
    }

    private static func l2(_ v: [Float]) -> [Float] {
        let n = sqrt(v.reduce(0) { $0 + $1 * $1 })
        return n > 0 ? v.map { $0 / n } : v
    }

    /// Embed an image file. Returns nil ONLY on a thrown/failed inference (runner → markDerivationFailed).
    public func embedImage(at url: URL) -> [Float]? {
        guard let m = model(resource: "MobileCLIPImage", cached: &imageModel) else { return nil }
        // ... build MLFeatureProvider from the image (CVPixelBuffer at the model's input size),
        //     predict, read the output MLMultiArray → [Float]. (Concrete I/O names come from the
        //     spike's model inspection; see docs/spikes/2026-06-11-mobileclip-coreml.md.)
        guard let raw = Self.predictVector(m, imageURL: url) else { return nil }
        return Self.l2(raw)
    }

    /// Embed a query string (query-time; not used by the background runner).
    public func embedText(_ s: String) -> [Float]? {
        guard let m = model(resource: "MobileCLIPText", cached: &textModel) else { return nil }
        let tokens = tokenizer.encode(s)
        guard let raw = Self.predictVector(m, tokens: tokens) else { return nil }
        return Self.l2(raw)
    }
}
```
> The two `predictVector` helpers (image pixel-buffer prep + token `MLMultiArray` prep → `m.prediction` → output `MLMultiArray` → `[Float]`) are implemented from the exact input/output feature names the spike recorded. Keep them private to `EmbedStage`.

```swift
/// Pure combination of the three lanes into one ordered hash list (the unit-tested heart).
public enum SearchRanker {
    public static func combine(structured: [String], text: [String],
                               semantic: [(hash: String, score: Float)], hasText: Bool) -> [String] {
        guard hasText else { return structured }     // empty box → filter set (already date-ordered)
        let allow = Set(structured)
        var out: [String] = []
        var seen = Set<String>()
        for h in text where allow.contains(h) && seen.insert(h).inserted { out.append(h) }
        for (h, _) in semantic where allow.contains(h) && seen.insert(h).inserted { out.append(h) }
        return out
    }
```

`SemanticIndex.swift`:

```swift
import Foundation
import Accelerate

/// In-memory brute-force cosine index over the current model's embeddings.
/// Vectors are L2-normalized at write time, so cosine == dot product.
public final class SemanticIndex: @unchecked Sendable {
    private let hashes: [String]
    private let matrix: [Float]      // count × dim, row-major, fp32
    public let dim: Int

    public init(catalog: Catalog, model: String) throws {
        let rows = try catalog.allEmbeddings(model: model)
        self.dim = rows.first?.dim ?? 0
        self.hashes = rows.map(\.hash)
        var m = [Float](); m.reserveCapacity(rows.count * dim)
        for r in rows where r.dim == dim { m.append(contentsOf: r.vector) }
        self.matrix = m
    }

    public var count: Int { hashes.count }

    /// Top-N by descending cosine. No hard threshold (CLIP cosine magnitudes are unreliable).
    public func query(_ q: [Float], topN: Int) -> [(hash: String, score: Float)] {
        guard dim > 0, q.count == dim, !hashes.isEmpty else { return [] }
        var scores = [Float](repeating: 0, count: hashes.count)
        // scores = matrix (count×dim) · q (dim) → count.  cblas row-major sgemv.
        matrix.withUnsafeBufferPointer { mb in
            q.withUnsafeBufferPointer { qb in
                cblas_sgemv(CblasRowMajor, CblasNoTrans, Int32(hashes.count), Int32(dim),
                            1, mb.baseAddress, Int32(dim), qb.baseAddress, 1, 0, &scores, 1)
            }
        }
        let n = min(topN, hashes.count)
        let top = scores.enumerated().sorted { $0.element > $1.element }.prefix(n)
        return top.map { (hashes[$0.offset], $0.element) }
    }
}
```

```swift
    private let derivationStages: [any DerivationStage] = [OCRDerivationStage(), EmbedStage()]

    private func drainDerivation() async {
        guard let lib = library else { return }
        for stage in derivationStages {
            if Task.isCancelled { break }
            let pending = (try? lib.catalog.pendingDerivation(stage: stage.id)) ?? []
            guard !pending.isEmpty else { continue }
            var progress = (try? lib.catalog.derivationProgress(stage: stage.id))
                ?? (done: 0, total: pending.count)
            derivationProgress = combinedProgress()
            for hash in pending {
                if Task.isCancelled { break }
                guard let url = goodCopyURL(forHash: hash, excluding: "") else { continue } // unreachable → skip
                let ok = await Task.detached(priority: .utility) { await stage.run(hash: hash, url: url, catalog: lib.catalog) }.value
                if ok { try? lib.catalog.markDerived(hash: hash, stage: stage.id); progress.done += 1 }
                else  { try? lib.catalog.markDerivationFailed(hash: hash, stage: stage.id) }
                derivationProgress = combinedProgress()
                await Task.yield()
            }
        }
        derivationProgress = nil
    }

    /// Combined remaining work across all stages (sidebar shows the sum).
    private func combinedProgress() -> (done: Int, total: Int)? {
        guard let lib = library else { return nil }
        var done = 0, total = 0
        for stage in derivationStages {
            if let p = try? lib.catalog.derivationProgress(stage: stage.id) { done += p.done; total += p.total }
        }
        return total > 0 ? (done, total) : nil
    }
```
